edge/workers/router.js
export default {
  async fetch(request, env) {
    // Edge routing logic
    const url = new URL(request.url);

    // Health check
    if (url.pathname === '/healthz') {
      return new Response('OK', { status: 200 });
    }

    // Route to backend
    return fetch(env.BACKEND_URL + url.pathname);
  }
}
